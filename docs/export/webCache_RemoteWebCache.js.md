/**
 * RemoteWebCache - Fetches a WhatsApp Web version index from a remote server
 * @param {object} options - options
 * @param {string} options.remotePath - Endpoint that should be used to fetch the version index. Use {version} as a placeholder for the version number.
 * @param {boolean} options.strict - If true, will throw an error if the requested version can&#x27;t be fetched. If false, will resolve to the latest version. Defaults to false.
 */
class RemoteWebCache extends WebCache {
    constructor(options &#x3D; {}) {
        super();

        if (!options.remotePath) throw new Error(&#x27;webVersionCache.remotePath is required when using the remote cache&#x27;);
        this.remotePath &#x3D; options.remotePath;
        this.strict &#x3D; options.strict || false;
    }

    async resolve(version) {
        const remotePath &#x3D; this.remotePath.replace(&#x27;{version}&#x27;, version);

        try {
            const cachedRes &#x3D; await fetch(remotePath);
            if (cachedRes.ok) {
                return cachedRes.text();
            }
        } catch (err) {
            console.error(&#x60;Error fetching version ${version} from remote&#x60;, err);
        }

        if (this.strict) throw new VersionResolveError(&#x60;Couldn&#x27;t load version ${version} from the archive&#x60;);
        return null;         
    }

    async persist() {
        // Nothing to do here
    }
}

module.exports &#x3D; RemoteWebCache;
