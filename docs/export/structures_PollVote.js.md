/**
 * Represents a Poll Vote on WhatsApp
 * @extends {Base}
 */
class PollVote extends Base {
    constructor(client, data) {
        super(client);

        if (data) this._patch(data);
    }

    _patch(data) {
        /**
         * The person who voted
         * @type {string}
         */
        this.voter &#x3D; data.sender;

        /**
         * The selected poll option(s)
         * If it&#x27;s an empty array, the user hasn&#x27;t selected any options on the poll,
         * may occur when they deselected all poll options
         * @type {SelectedPollOption[]}
         */
        if (data.selectedOptionLocalIds.length > 0) {
            if(data.parentMessage) { // temporary failsafe
                this.selectedOptions &#x3D; data.selectedOptionLocalIds.map((e) &#x3D;> ({
                    name: data.parentMessage.pollOptions.find((x) &#x3D;> x.localId &#x3D;&#x3D;&#x3D; e).name,
                    localId: e
                }));
            } else {
                this.selectedOptions &#x3D; data.selectedOptionLocalIds.map((e) &#x3D;> ({
                    name: undefined,
                    localId: e
                }));
            }
        } else {
            this.selectedOptions &#x3D; [];
        }
        
        /**
         * Timestamp the option was selected or deselected at
         * @type {number}
         */
        this.interractedAtTs &#x3D; data.senderTimestampMs;

        /**
         * The poll creation message associated with the poll vote
         * @type {Message}
         */
        this.parentMessage &#x3D; new Message(this.client, data.parentMessage);

        /**
         * The poll creation message id
         * @type {Object}
         */
        this.parentMsgKey &#x3D;  data.parentMsgKey;

        return super._patch(data);
    }
}

module.exports &#x3D; PollVote;
