&#x27;use strict&#x27;;

const Base &#x3D; require(&#x27;./Base&#x27;);
const Message &#x3D; require(&#x27;./Message&#x27;);

/**
 * Represents a Status/Story on WhatsApp
 * @extends {Base}
 */
class Broadcast extends Base {
    constructor(client, data) {
        super(client);

        if (data) this._patch(data);
    }

    _patch(data) {
        /**
         * ID that represents the chat
         * @type {object}
         */
        this.id &#x3D; data.id;

        /**
         * Unix timestamp of last status
         * @type {number}
         */
        this.timestamp &#x3D; data.t;

        /**
         * Number of available statuses
         * @type {number}
         */
        this.totalCount &#x3D; data.totalCount;

        /**
         * Number of not viewed
         * @type {number}
         */
        this.unreadCount &#x3D; data.unreadCount;

        /**
         * Messages statuses
         * @type {Message[]}
         */
        this.msgs &#x3D; data.msgs?.map(msg &#x3D;> new Message(this.client, msg));

        return super._patch(data);
    }

    /**
     * Returns the Chat this message was sent in
     * @returns {Promise&lt;Chat>}
     */
    getChat() {
        return this.client.getChatById(this.id._serialized);
    }

    /**
     * Returns the Contact this message was sent from
     * @returns {Promise&lt;Contact>}
     */
    getContact() {
        return this.client.getContactById(this.id._serialized);
    }

}

module.exports &#x3D; Broadcast;
