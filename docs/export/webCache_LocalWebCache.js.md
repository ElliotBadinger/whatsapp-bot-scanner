/**
 * LocalWebCache - Fetches a WhatsApp Web version from a local file store
 * @param {object} options - options
 * @param {string} options.path - Path to the directory where cached versions are saved, default is: &quot;./.wwebjs_cache/&quot; 
 * @param {boolean} options.strict - If true, will throw an error if the requested version can&#x27;t be fetched. If false, will resolve to the latest version.
 */
class LocalWebCache extends WebCache {
    constructor(options &#x3D; {}) {
        super();

        this.path &#x3D; options.path || &#x27;./.wwebjs_cache/&#x27;;
        this.strict &#x3D; options.strict || false;
    }

    async resolve(version) {
        const filePath &#x3D; path.join(this.path, &#x60;${version}.html&#x60;);
        
        try {
            return fs.readFileSync(filePath, &#x27;utf-8&#x27;);
        }
        catch (err) {
            if (this.strict) throw new VersionResolveError(&#x60;Couldn&#x27;t load version ${version} from the cache&#x60;);
            return null;
        }
    }

    async persist(indexHtml, version) {
        // version &#x3D; (version+&#x27;&#x27;).replace(/[^0-9.]/g,&#x27;&#x27;);
        const filePath &#x3D; path.join(this.path, &#x60;${version}.html&#x60;);
        fs.mkdirSync(this.path, { recursive: true });
        fs.writeFileSync(filePath, indexHtml);
    }
}

module.exports &#x3D; LocalWebCache;
